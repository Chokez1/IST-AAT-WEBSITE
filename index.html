# STEP 12 : SAMPLING TECHNIQUES COMPARISON

# Prepare training data as DataFrame for sampling
X_train_df = pd.DataFrame(X_train_scaled).reset_index(drop=True)
y_train_sr = y_train.reset_index(drop=True)

FRAC = 0.6          # use 60% of training data as sample
all_results = []     # collect results here
np.random.seed(42)

# --- Helper: train both models and store metrics ---
def train_and_evaluate(sampling_name, X_sample, y_sample):
    """Train KNN + Naive Bayes on the sample, evaluate on test set."""

    # KNN
    knn = KNeighborsClassifier(n_neighbors=5)
    knn.fit(X_sample, y_sample)
    kp = knn.predict(X_test_scaled)

    all_results.append({
        "Sampling": sampling_name,
        "Model": "KNN",
        "Accuracy":  round(accuracy_score(y_test, kp), 4),
        "Precision": round(precision_score(y_test, kp), 4),
        "Recall":    round(recall_score(y_test, kp), 4),
        "F1":        round(f1_score(y_test, kp), 4),
    })

    # Naive Bayes
    nb = GaussianNB()
    nb.fit(X_sample, y_sample)
    nbp = nb.predict(X_test_scaled)

    all_results.append({
        "Sampling": sampling_name,
        "Model": "Naive Bayes",
        "Accuracy":  round(accuracy_score(y_test, nbp), 4),
        "Precision": round(precision_score(y_test, nbp), 4),
        "Recall":    round(recall_score(y_test, nbp), 4),
        "F1":        round(f1_score(y_test, nbp), 4),
    })

# ---------- 1. Simple Random Sampling ----------
print("\n1) Simple Random Sampling")
n_sample = int(len(X_train_df) * FRAC)
rand_idx = np.random.choice(len(X_train_df), size=n_sample, replace=False)
X_srs = X_train_df.iloc[rand_idx]
y_srs = y_train_sr.iloc[rand_idx]
print(f"   Sampled {len(X_srs)} rows randomly from {len(X_train_df)}")
train_and_evaluate("Simple Random", X_srs, y_srs)

# ---------- 2. Stratified Sampling ----------
print("\n2) Stratified Sampling")
sss = StratifiedShuffleSplit(n_splits=1, train_size=FRAC, random_state=42)
for strat_idx, _ in sss.split(X_train_df, y_train_sr):
    X_strat = X_train_df.iloc[strat_idx]
    y_strat = y_train_sr.iloc[strat_idx]
print(f"   Sampled {len(X_strat)} rows preserving class proportions")
train_and_evaluate("Stratified", X_strat, y_strat)

# ---------- 3. Systematic Sampling ----------
print("\n3) Systematic Sampling")
step = max(1, int(1 / FRAC))
sys_idx = np.arange(0, len(X_train_df), step)
X_sys = X_train_df.iloc[sys_idx]
y_sys = y_train_sr.iloc[sys_idx]
print(f"   Every {step}-th row selected → {len(X_sys)} rows")
train_and_evaluate("Systematic", X_sys, y_sys)

# ---------- 4. Cluster Sampling ----------
print("\n4) Cluster Sampling")
n_clusters = 6
km = KMeans(n_clusters=n_clusters, random_state=42, n_init=10)
cluster_labels = km.fit_predict(X_train_df)
chosen_clusters = np.random.choice(n_clusters, size=4, replace=False)
mask = np.isin(cluster_labels, chosen_clusters)
X_clust = X_train_df[mask]
y_clust = y_train_sr[mask]
print(f"   {n_clusters} clusters formed, {len(chosen_clusters)} selected → {len(X_clust)} rows")
train_and_evaluate("Cluster", X_clust, y_clust)
